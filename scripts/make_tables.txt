create table t_date_dim (
	date_key INT PRIMARY KEY NOT NULL,
	day_of_the_week VARCHAR(15),
	month_ VARCHAR(15),
	year_ VARCHAR(15),
	weekend Boolean,
	holiday Boolean,
	holiday_name Varchar(50)
)


create table t_location_dim (
	location_key INT PRIMARY KEY NOT NULL,
	location_name VARCHAR(50),
	longitude float,
	latitude float,
	neighborhood VARCHAR(50),
	city VARCHAR(50),
	crime_rate float
)


CREATE TABLE t_crime_dim(
   crime_key serial PRIMARY KEY,
   crime_report_time TIME,
   crime_start_time TIME,
   crime_end_time TIME,
   crime_type VARCHAR(150),
   crime_category VARCHAR(50),
   crime_severity_index float
);
